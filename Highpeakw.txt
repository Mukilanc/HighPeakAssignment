#include<stdio.h>
#include<math.h>

int asc(int array[],int N){
    int temp,i,j;
    for(i=0;i<N;i++){
        //int swapped=0}
        for(j=i+1;j<N;j++){
        if(array[i]>array[j])
            {
                temp=array[i];
                array[i]=array[j];
                array[j]=temp;
                //swapped=i
            }
        }
    }
}

int mini(int r, int s){
    if(r>s)
        return s;
    else 
        return r;
}
int minDifference(int array[],int N,int M){
    int result = 2147483647;
    asc(array, N);
    for(int i=0;i<=N-M;i++)
        {
            result = mini(result,array[i+M-1]-array[i]);
        }
        return result;
}
int findele(int res,int array[],int N,int M){
    int result=2147483647;
    for(int i=0;i<=N-M;i++)
        {
            result = mini(result,array[i+M-1]-array[i]);
            if(res==result){
            return i;
        }
        return 0;
        }
}
